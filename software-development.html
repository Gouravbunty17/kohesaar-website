import { Database } from '@/lib/db';
import { validateUser } from '@/utils';

// User service with type safety
export class UserService {
  private db: Database;

  async create(data: UserInput) {
    const validated = validateUser(data);
    return this.db.users.create({
      data: validated,
    });
  }
}
